```python
import numpy as np

# Parameters
k = 1000.0  # W/(m*K)
hP_kA = 25.0  # m^-2
T_inf = 20.0  # °C
L = 0.5  # m
T0 = 100.0  # °C
T1 = 200.0  # °C

# Discretization
N = 100  # number of control volumes
dx = L / N

# Coefficients
aW = k / dx
aE = k / dx
aP0 = aW + aE + hP_kA * dx
aP = aW + aE + hP_kA * dx

# Initialize temperature array
T = np.zeros(N + 2)
T[0] = T0
T[-1] = T1

# Coefficient matrix and right-hand side vector
A = np.zeros((N, N))
b = np.zeros(N)

# Fill the coefficient matrix and RHS vector
for i in range(N):
    if i == 0:
        A[i, i] = aP0
        A[i, i + 1] = -aE
        b[i] = hP_kA * dx * T_inf + aW * T0
    elif i == N - 1:
        A[i, i - 1] = -aW
        A[i, i] = aP0
        b[i] = hP_kA * dx * T_inf + aE * T1
    else:
        A[i, i - 1] = -aW
        A[i, i] = aP
        A[i, i + 1] = -aE
        b[i] = hP_kA * dx * T_inf

# Solve the linear system
T_internal = np.linalg.solve(A, b)

# Combine the boundary and internal temperatures
T[1:-1] = T_internal

# Save the temperature distribution
np.save('T.npy', T)
```
