```python
import numpy as np
import math

# Parameters
L = 2 * math.pi
nu = 0.5
dt = 0.01
dx = dt / nu
N = math.ceil(L / dx)
x = np.linspace(0, L, N)
T = 500

# Initial condition
u = np.sin(x) + 0.5 * np.sin(0.5 * x)

# Time integration using Lax method
for _ in range(T):
    F = 0.5 * u**2
    u_p = np.roll(u, -1)
    u_m = np.roll(u, 1)
    F_p = np.roll(F, -1)
    F_m = np.roll(F, 1)
    u = (u_p + u_m) / 2 - (dt / (2 * dx)) * (F_p - F_m)

# Save the final solution
np.save('u.npy', u)
```
