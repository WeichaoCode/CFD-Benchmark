```python
import numpy as np
import matplotlib.pyplot as plt

# Problem parameters
L = 2.0  # Domain length
u = 0.2  # Fluid velocity
T = 2.5  # Total simulation time
m = 0.5  # Gaussian peak location
s = 0.1  # Gaussian width

# Discretization parameters
nx = 200  # Number of spatial points
nt = 500  # Number of time points

# Grid generation
dx = L / (nx - 1)
x = np.linspace(0, L, nx)
dt = T / (nt - 1)
t = np.linspace(0, T, nt)

# Initialize solution array
phi = np.zeros(nx)

# Initial condition (Gaussian profile)
phi = np.exp(-((x - m)/s)**2)

# Finite Volume Method (Upwind scheme)
for n in range(1, nt):
    phi_old = phi.copy()
    
    # Upwind flux calculation
    for i in range(1, nx-1):
        # Upwind discretization
        phi[i] = phi_old[i] - (u * dt / dx) * (phi_old[i] - phi_old[i-1])
    
    # Enforce boundary conditions
    phi[0] = 0
    phi[-1] = 0

# Save final solution
np.save('phi.npy', phi)
```
