```python
import numpy as np

# Parameters
nx, ny = 41, 41
nt = 10
dx = 2 / (nx - 1)
dy = 2 / (ny - 1)
rho = 1
nu = 0.1
F = 1
dt = 0.01

# Initialize fields
u = np.zeros((ny, nx))
v = np.zeros((ny, nx))
p = np.zeros((ny, nx))
b = np.zeros((ny, nx))

# Function to apply periodic boundary conditions in x-direction
def apply_periodic_bc(u, v, p):
    u[:, 0] = u[:, -2]
    u[:, -1] = u[:, 1]
    v[:, 0] = v[:, -2]
    v[:, -1] = v[:, 1]
    p[:, 0] = p[:, -2]
    p[:, -1] = p[:, 1]

# Function to apply no-slip boundary conditions in y-direction
def apply_no_slip_bc(u, v):
    u[0, :] = 0
    u[-1, :] = 0
    v[0, :] = 0
    v[-1, :] = 0

# Function to apply Neumann boundary conditions for pressure in y-direction
def apply_neumann_bc(p):
    p[0, :] = p[1, :]
    p[-1, :] = p[-2, :]

# Pressure Poisson equation solver using Jacobi iteration
def pressure_poisson(p, b, dx, dy):
    pn = np.empty_like(p)
    for _ in range(50):  # Number of iterations
        pn = p.copy()
        p[1:-1, 1:-1] = (((pn[1:-1, 2:] + pn[1:-1, :-2]) * dy**2 +
                          (pn[2:, 1:-1] + pn[:-2, 1:-1]) * dx**2) /
                         (2 * (dx**2 + dy**2)) -
                         dx**2 * dy**2 / (2 * (dx**2 + dy**2)) * b[1:-1, 1:-1])
        apply_periodic_bc(u, v, p)
        apply_neumann_bc(p)

# Time-stepping loop
for n in range(nt):
    un = u.copy()
    vn = v.copy()

    # Build up the RHS of the pressure Poisson equation
    b[1:-1, 1:-1] = (rho * (1 / dt *
                            ((un[1:-1, 2:] - un[1:-1, :-2]) / (2 * dx) +
                             (vn[2:, 1:-1] - vn[:-2, 1:-1]) / (2 * dy)) -
                            ((un[1:-1, 2:] - un[1:-1, :-2]) / (2 * dx))**2 -
                            2 * ((un[2:, 1:-1] - un[:-2, 1:-1]) / (2 * dy) *
                                 (vn[1:-1, 2:] - vn[1:-1, :-2]) / (2 * dx)) -
                            ((vn[2:, 1:-1] - vn[:-2, 1:-1]) / (2 * dy))**2))

    apply_periodic_bc(u, v, p)
    apply_neumann_bc(p)

    # Solve for pressure
    pressure_poisson(p, b, dx, dy)

    # Update velocity fields
    u[1:-1, 1:-1] = (un[1:-1, 1:-1] -
                     un[1:-1, 1:-1] * dt / dx *
                     (un[1:-1, 1:-1] - un[1:-1, :-2]) -
                     vn[1:-1, 1:-1] * dt / dy *
                     (un[1:-1, 1:-1] - un[:-2, 1:-1]) -
                     dt / (2 * rho * dx) * (p[1:-1, 2:] - p[1:-1, :-2]) +
                     nu * (dt / dx**2 *
                           (un[1:-1, 2:] - 2 * un[1:-1, 1:-1] + un[1:-1, :-2]) +
                           dt / dy**2 *
                           (un[2:, 1:-1] - 2 * un[1:-1, 1:-1] + un[:-2, 1:-1])) +
                     F * dt)

    v[1:-1, 1:-1] = (vn[1:-1, 1:-1] -
                     un[1:-1, 1:-1] * dt / dx *
                     (vn[1:-1, 1:-1] - vn[1:-1, :-2]) -
                     vn[1:-1, 1:-1] * dt / dy *
                     (vn[1:-1, 1:-1] - vn[:-2, 1:-1]) -
                     dt / (2 * rho * dy) * (p[2:, 1:-1] - p[:-2, 1:-1]) +
                     nu * (dt / dx**2 *
                           (vn[1:-1, 2:] - 2 * vn[1:-1, 1:-1] + vn[1:-1, :-2]) +
                           dt / dy**2 *
                           (vn[2:, 1:-1] - 2 * vn[1:-1, 1:-1] + vn[:-2, 1:-1])))

    apply_periodic_bc(u, v, p)
    apply_no_slip_bc(u, v)

# Save the final solution
np.save('u.npy', u)
np.save('v.npy', v)
np.save('p.npy', p)
```

This code implements a finite difference method to solve the given Navier-Stokes equations with the specified boundary and initial conditions. The solution is saved at the final time step in `.npy` format for the velocity components `u`, `v`, and the pressure `p`.
