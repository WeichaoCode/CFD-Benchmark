```python
import numpy as np
import math

# Parameters
L = 2 * np.pi
nu = 0.5
dt = 0.01
dx = dt / nu
N = math.ceil(L / dx)
x = np.linspace(0, L, N)
T = 500

# Initial condition
u = np.sin(x) + 0.5 * np.sin(0.5 * x)

# Lax method time integration
for _ in range(T):
    u_new = 0.5 * (np.roll(u, -1) + np.roll(u, 1)) - (dt / (2 * dx)) * (0.5 * np.roll(u, -1)**2 - 0.5 * np.roll(u, 1)**2)
    u = u_new

# Save the final solution
np.save('u.npy', u)
```
